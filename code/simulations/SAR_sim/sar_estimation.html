function estimates =  est_SAR(Y, X, W)

    %normalize W
    for i=1:size(W,1)
        if sum(W(i,:))~=0
            W(i,:) = W(i,:)/sum(W(i,:));
        end
    end

    rng(31, 'twister');
    info.lflag = 0; % use full lndet no approximation
    estimates = sar( Y, X, W, info);

end
